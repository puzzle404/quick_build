```ruby
class Expense < ApplicationRecord
  belongs_to :project
  belongs_to :project_stage, optional: true
  belongs_to :author, class_name: "User"

  has_one_attached :receipt

  enum :category, { labor: 0, materials_misc: 1, rentals: 2, other: 3 }

  validates :amount_cents, presence: true, numericality: { greater_than: 0 }
  validates :incurred_on, presence: true
  validate :stage_belongs_to_same_project

  scope :for_project, ->(project_id) { where(project_id: project_id) }
  scope :for_stage,   ->(stage_id)   { where(project_stage_id: stage_id) }
  scope :recent_first, -> { order(incurred_on: :desc, id: :desc) }

  def amount
    amount_cents / 100.0
  end

  private

  def stage_belongs_to_same_project
    return if project_stage.blank?
    return if project_stage.project_id == project_id

    errors.add(:project_stage, "debe pertenecer al mismo proyecto")
  end
end
```

```ruby
def predecessor_belongs_to_same_project
  return if predecessor.blank?
  return if predecessor.project_id == project_id

  errors.add(:predecessor, "debe pertenecer al mismo proyecto")
end

def predecessor_must_not_be_self
  return if predecessor.blank?
  return unless persisted? && predecessor_id == id

  errors.add(:predecessor, "no puede ser ella misma")
end

def predecessor_must_not_create_cycle
  return if predecessor.blank?

  visited = Set.new([id].compact)
  cursor  = predecessor
  while cursor
    if visited.include?(cursor.id)
      errors.add(:predecessor, "genera un ciclo de dependencias")
      return
    end
    visited << cursor.id
    cursor = cursor.predecessor
  end
end

def start_after_predecessor_end
  return if predecessor.blank? || start_date.blank? || predecessor.end_date.blank?
  return if start_date >= predecessor.end_date

  errors.add(:start_date,
             "no puede ser anterior al fin de la etapa predecesora (#{predecessor.end_date})")
end
```

```ruby
module Projects
  class ProgressCalculator
    def initialize(project)
      @project = project
    end

    def percent
      total_weight = 0
      weighted_sum = 0.0

      root_stages.each do |stage|
        days = duration_days(stage)
        next if days.zero?

        total_weight += days
        weighted_sum += days * (stage.progress.to_i / 100.0)
      end

      return 0 if total_weight.zero?

      ((weighted_sum / total_weight) * 100).round
    end

    private

    def root_stages
      @project.project_stages.where(parent_id: nil)
    end

    def duration_days(stage)
      return 0 if stage.start_date.blank? || stage.end_date.blank?

      (stage.end_date - stage.start_date).to_i
    end
  end
end
```

```ruby
class Constructors::ExpensesController < Constructors::BaseController
  before_action :set_project
  before_action :set_stage, only: [:create]
  before_action :set_expense, only: [:destroy]

  def create
    @expense = @project.expenses.new(expense_params)
    @expense.project_stage = @stage if @stage
    @expense.author = current_user

    authorize @expense

    if @expense.save
      redirect_back_or_to redirect_target, notice: "Gasto registrado."
    else
      redirect_back_or_to redirect_target, alert: @expense.errors.full_messages.to_sentence
    end
  end

  def destroy
    authorize @expense
    @expense.destroy
    redirect_back_or_to constructors_project_path(@project), notice: "Gasto eliminado."
  end

  private

  def set_project
    @project = current_user.owned_projects.find(params[:project_id])
  end

  def set_stage
    return unless params[:stage_id]

    @stage = @project.project_stages.find(params[:stage_id])
  end

  def set_expense
    @expense = @project.expenses.find(params[:id])
  end

  def redirect_target
    if @stage
      constructors_project_stage_path(@project, @stage)
    else
      constructors_project_path(@project)
    end
  end

  def expense_params
    params.require(:expense).permit(
      :amount_cents, :currency, :category, :incurred_on, :description, :receipt
    )
  end
```

```ruby
class Constructors::NotesController < Constructors::BaseController
  before_action :set_project
  before_action :set_noteable, only: [:create]
  before_action :set_note,     only: [:destroy]

  def create
    @note = @noteable.notes.new(note_params.merge(author: current_user))

    authorize @note

    if @note.save
      redirect_back_or_to noteable_path(@noteable), notice: "Nota guardada."
    else
      redirect_back_or_to noteable_path(@noteable), alert: @note.errors.full_messages.to_sentence
    end
  end

  def destroy
    authorize @note
    parent = @note.noteable
    @note.destroy
    redirect_back_or_to noteable_path(parent), notice: "Nota eliminada."
  end

  private

  def set_project
    @project = current_user.owned_projects.find(params[:project_id])
  end

  def set_noteable
    @noteable = if params[:stage_id]
                  @project.project_stages.find(params[:stage_id])
                else
                  @project
                end
  end

  def set_note
    scope = if params[:stage_id]
              @project.project_stages.find(params[:stage_id]).notes
            else
              @project.notes
            end
    @note = scope.find(params[:id])
  end

  def noteable_path(noteable)
    case noteable
    when Project       then constructors_project_path(noteable)
    when ProjectStage  then constructors_project_stage_path(@project, noteable)
    end
  end

  def note_params
    params.require(:note).permit(:title, :body)
  end
end
```

```ruby
require "net/http"
require "json"

module External
  class ExchangeRatesFetcher
    URL = URI("https://dolarapi.com/v1/dolares").freeze
    CACHE_KEY      = "fx:dolarapi".freeze
    LAST_KNOWN_KEY = "fx:dolarapi:last_known".freeze
    TIMEOUT_S      = 5

    def call
      Rails.cache.fetch(CACHE_KEY, expires_in: 1.hour) do
        result = fetch_remote
        Rails.cache.write(LAST_KNOWN_KEY, result) if result[:stale] == false
        result
      end
    end

    private

    def fetch_remote
      response = Net::HTTP.start(URL.host, URL.port, use_ssl: true, open_timeout: TIMEOUT_S, read_timeout: TIMEOUT_S) do |http|
        http.get(URL.request_uri)
      end

      raise "HTTP #{response.code}" unless response.code.to_i == 200

      parse(JSON.parse(response.body))
    rescue StandardError => e
      Rails.logger.warn("ExchangeRatesFetcher failed: #{e.message}")
      last = Rails.cache.read(LAST_KNOWN_KEY) || { oficial: {}, blue: {} }
      last.merge(stale: true)
    end

    def parse(payload)
      oficial = payload.find { |r| r["casa"] == "oficial" } || {}
      blue    = payload.find { |r| r["casa"] == "blue" }    || {}

      {
        oficial: extract(oficial),
        blue:    extract(blue),
        stale:   false
      }
    end

    def extract(row)
      return {} if row.empty?

      {
        compra: row["compra"],
        venta:  row["venta"],
        fecha:  row["fechaActualizacion"]
      }
    end
  end
end
```

```ruby
require "net/http"
require "json"

module External
  class WeatherFetcher
    BASE     = "https://api.openweathermap.org/data/2.5/forecast".freeze
    TIMEOUT  = 5

    def initialize(lat:, lng:)
      @lat = lat
      @lng = lng
    end

    def call
      return nil if @lat.blank? || @lng.blank?

      Rails.cache.fetch(cache_key, expires_in: 1.hour) do
        result = fetch_remote
        Rails.cache.write(last_known_key, result) if result[:stale] == false
        result
      end
    end

    private

    def cache_key      = "weather:#{lat_r}:#{lng_r}"
    def last_known_key = "weather:#{lat_r}:#{lng_r}:last_known"

    def lat_r = @lat.to_f.round(2)
    def lng_r = @lng.to_f.round(2)

    def fetch_remote
      uri = URI(BASE)
      uri.query = URI.encode_www_form(
        lat: lat_r, lon: lng_r,
        units: "metric", lang: "es",
        appid: ENV.fetch("OPENWEATHER_API_KEY", "")
      )

      response = Net::HTTP.start(uri.host, uri.port, use_ssl: true, open_timeout: TIMEOUT, read_timeout: TIMEOUT) do |http|
        http.get(uri.request_uri)
      end

      raise "HTTP #{response.code}" unless response.code.to_i == 200

      parse(JSON.parse(response.body))
    rescue StandardError => e
      Rails.logger.warn("WeatherFetcher failed: #{e.message}")
      last = Rails.cache.read(last_known_key) || { days: [] }
      last.merge(stale: true)
    end

    def parse(payload)
      slots = payload["list"] || []
      by_day = slots.group_by { |s| Time.zone.at(s["dt"]).to_date }

      days = by_day.sort.first(5).map do |date, slots_in_day|
        rep = slots_in_day.find { |s| Time.zone.at(s["dt"]).hour.between?(11, 14) } || slots_in_day.first
        {
          date:        date,
          max:         slots_in_day.map { |s| s.dig("main", "temp_max") }.compact.max,
          min:         slots_in_day.map { |s| s.dig("main", "temp_min") }.compact.min,
          icon:        rep.dig("weather", 0, "icon"),
          description: rep.dig("weather", 0, "description")
        }
      end

      current = days.first || {}
      { current: current, days: days, stale: false }
    end
  end
end
```
